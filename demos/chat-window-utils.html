ChatWindowUtils 使用事例

概述

ChatWindowUtils 是 Chat 组件的共享工具函数集，9 个纯函数。提供安全的调用包装、Vue API 检测、组件模块查找、模板加载、颜色计算等跨组件复用能力。

全部 9 个工具函数

safeCall

function safeCall(fn, fallbackValue) {
  try { return fn() }
  catch (_) { return fallbackValue ?? null }
}

同步安全调用，异常时返回 fallbackValue

safeCallAsync

async function safeCallAsync(fn, fv) {
  try { return await fn() }
  catch (_) { return fv ?? null }
}

异步安全调用，异常时返回 fallbackValue

getVueApi

function getVueApi(Vue) {
  if (!Vue?.createApp || !Vue?.defineComponent
      || !Vue?.ref || !Vue?.onMounted)
    return null
  return { createApp, defineComponent, ref, onMounted }
}

检测 Vue 3 可用性，提取核心 API

canUseVueTemplate

function canUseVueTemplate(Vue) {
  if (typeof Vue?.compile !== 'function')
    return false
  return safeCall(() => {
    Function('return 1')(); return true
  }, false)
}

检测 template 编译 + CSP 是否允许 Function 构造器

getComponentModule

function getComponentModule(name) {
  return window.PetManager?.Components
    ?.[name] || null
}

从全局命名空间获取组件模块

loadTemplateIfAvailable

async function loadTemplateIfAvailable(mod) {
  if (!mod?.loadTemplate) return ''
  return await safeCallAsync(
    () => mod.loadTemplate(), '')
}

安全加载组件 HTML 模板

toRgbFromHex

function toRgbFromHex(hex) {
  const m = hex.match(/^#([0-9a-fA-F]{6})$/)
  if (!m) return null
  return { r: parseInt(...), g: ..., b: ... }
}

6 位 hex → { r, g, b } 对象

clampInt

function clampInt(n, min, max) {
  const x = Math.round(Number(n))
  if (!Number.isFinite(x)) return min
  return Math.min(Math.max(x, min), max)
}

整数钳制，非数字返回 min

shadeHexColor

function shadeHexColor(hex, ratio) {
  const rgb = toRgbFromHex(hex)
  if (!rgb) return null
  const t = ratio < 0 ? 0 : 255
  // 线性插值: (t - c) * |ratio| + c
  return '#' + [r,g,b].map(...).join('')
}

颜色加深/减淡，ratio < 0 为加深

ChatWindow 中的使用

const _u = window.PetManager?.Components?.ChatWindowUtils || {}

// 所有子组件通过 _u 引用工具函数
const safeCall = _u.safeCall || function(fn, fv) { ... }
const getVueApi = _u.getVueApi || null
const canUseVueTemplate = _u.canUseVueTemplate || function() { ... }
const toRgbFromHex = _u.toRgbFromHex || function() { ... }

// 主题更新管线: hex → rgb → shadeHex → CSS vars
const mainColor = this.getMainColorFromGradient(gradient)
const rgb = toRgbFromHex(mainColor)
const hoverColor = shadeHexColor(mainColor, -0.08)
